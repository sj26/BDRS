# Setup mod_jk for the BDRS
<IfModule mod_jk.c>
    JkWorkersFile /etc/tomcat6/workers.properties
    JkLogFile /var/log/tomcat6/mod_jk.log
    JkShmFile /var/log/tomcat6/shm

    JkLogLevel error

    JkMount /BDRS/* ajp13
    # Remove this after configuration is complete.
    JkMount /manager/* ajp13
</IfModule>
